// Render events
function renderEvents(filter = 'all') {
const container = document.getElementById('eventsContainer');
container.innerHTML = '';

const filteredEvents = filter === 'all'
? historyEvents
: historyEvents.filter(e => e.category === filter);

filteredEvents.forEach((event, index) => {
const isLeft = index % 2 === 0;
const categoryClass = `category-${event.category}`;

const eventHTML = `
<div class="event-card flex flex-col md:flex-row items-center w-full ${isLeft ? 'md:flex-row' : 'md:flex-row-reverse'}" data-category="${event.category}">
<div class="w-full md:w-1/2 ${isLeft ? 'md:pr-12 md:text-right' : 'md:pl-12 md:text-left'} mb-4 md:mb-0">
<div class="bg-white/5 backdrop-blur-sm rounded-2xl p-6 border border-white/10 hover:border-white/30 transition-all cursor-pointer group ${categoryClass}" onclick="openModal(${event.id})">
<div class="flex items-center ${isLeft ? 'md:justify-end' : 'md:justify-start'} mb-3 space-x-2">
<span class="date-badge px-3 py-1 rounded-full text-xs font-bold text-white">${event.date}</span>
<span class="text-xs text-gray-500 uppercase tracking-wider">${event.category}</span>
</div>
<h3 class="text-xl font-bold text-white mb-2 group-hover:text-red-400 transition-colors">${event.title}</h3>
<p class="text-gray-400 text-sm leading-relaxed mb-4">${event.description}</p>
<div class="image-hover rounded-lg overflow-hidden h-48 relative">

<div class="absolute inset-0 bg-gradient-to-t from-black/60 to-transparent opacity-0 group-hover:opacity-100 transition-opacity flex items-end p-4">

Read more <i data-lucide="arrow-right" class="w-4 h-4 ml-2"></i>
</span>
</div>
</div>
</div>
</div>

<div class="absolute left-1/2 transform -translate-x-1/2 w-4 h-4 bg-white rounded-full border-4 border-red-600 z-10 hidden md:block"></div>

<div class="w-full md:w-1/2 ${isLeft ? 'md:pl-12' : 'md:pr-12'} hidden md:block">

});

// Re-initialize icons for new content
lucide.createIcons();

// Trigger animations
setTimeout(() => {
observeEvents();
}, 100);
}

// Intersection Observer for animations
function observeEvents() {
const observer = new IntersectionObserver((entries) => {
entries.forEach(entry => {
if (entry.isIntersecting) {
entry.target.classList.add('visible');
}
});
}, { threshold: 0.1 });

document.querySelectorAll('.event-card').forEach(card => {
observer.observe(card);
});
}

// Filter functionality
document.querySelectorAll('.filter-btn').forEach(btn => {
btn.addEventListener('click', function() {
document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));
this.classList.add('active');
renderEvents(this.dataset.filter);
});
});

// Animate statistics on scroll
function animateStats() {
const stats = [
{ id: 'stat1', target: 4000, suffix: '+' },
{ id: 'stat2', target: 3, suffix: '' },
{ id: 'stat3', target: 4, suffix: '' },
{ id: 'stat4', target: 30, suffix: '+' }
];

stats.forEach(stat => {
const element = document.getElementById(stat.id);
let current = 0;
const increment = stat.target / 50;
const timer = setInterval(() => {
current += increment;
if (current >= stat.target) {
current = stat.target;
clearInterval(timer);
}
element.textContent = Math.floor(current) + stat.suffix;
}, 30);
});
}

// Observe stats section
const statsObserver = new IntersectionObserver((entries) => {
entries.forEach(entry => {
if (entry.isIntersecting) {
animateStats();
statsObserver.unobserve(entry.target);
}
});
}, { threshold: 0.5 });

// Initialize
document.addEventListener('DOMContentLoaded', () => {
renderEvents();
statsObserver.observe(document.querySelector('#stat1').parentElement.parentElement);

// Close modal on escape key
document.addEventListener('keydown', (e) => {
if (e.key === 'Escape') closeModal();
});
});

// Smooth scroll for navigation
document.querySelectorAll('a[href^="#"]').forEach(anchor => {
anchor.addEventListener('click', function (e) {
e.preventDefault();
const target = document.querySelector(this.getAttribute('href'));
if (target) {
target.scrollIntoView({ behavior: 'smooth', block: 'start' });
}
});
});
</script>

This website features:

Design Elements:

           Color Palette: Inspired by the Kurdish flag (red, white, green) with gold accents representing the sun emblem

           Typography: Mix of Cinzel (elegant, historical feel), Inter (modern readability), and Noto Naskh Arabic (for Kurdish script)

           Dark Theme: Professional, museum-like aesthetic that makes content pop

Key Features:

1.        Interactive Timeline: Scroll-triggered animations, alternating left/right layout on desktop

2.        Filtering System: Filter events by historical periods (Ancient, Medieval, Ottoman, Modern, Contemporary)

3.        Detailed Modals: Click any event for expanded information with historical context

4.        Four Regions Section: Interactive cards explaining Bakr, Başr, Rojava, and Rojhilat

5.        Cultural Heritage: Showcasing music (Dengbj), Newroz, and language

6.        Animated Statistics: Counter animations when scrolling into view

7.        Responsive Design: Fully functional on mobile and desktop

Content Coverage:

           Ancient era: Mittani Kingdom, Medes, Xenophons accounts

           Medieval: Islamic conquest, Saladins rise

           Ottoman: Sharafnama, fall of emirates, Sheikh Ubeydullah

           Modern: Treaty of Svres, Sheikh Said rebellion, Republic of Mahabad

           Contemporary: PKK foundation, Anfal Campaign, ISIS war

The design uses sophisticated glass-morphism effects, gradient accents, and smooth transitions to create a premium historical archive feel. All images are sourced from Unsplash with relevant historical/landscape photography.
